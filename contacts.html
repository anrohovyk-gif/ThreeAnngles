function calculate() { let days = parseFloat(document.getElementById("days").value); let people = parseFloat(document.getElementById("people").value); let resultDiv = document.getElementById("result"); if (days > 0 && people > 0) { let dailyRate = 100; // Додаткові витрати на день let total = days * dailyRate * people; resultDiv.innerHTML = "Орієнтовні додаткові витрати: " + total + " $"; } else { resultDiv.innerText = "Будь ласка, введіть коректні дані."; } }
